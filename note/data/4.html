冗余索引是指多个索引的前缀列相同 或是在联合索引中包含了主键的索引 下面这个例子中key(name,id)就是一个冗余索引 create table test( id int not null primary key, name varchar(10) not null, title varchar(50) not null, key(name,id) )engine=innodb; 索引的维护及优化 --- 查找重复及冗余索引 select a.table_schema as '数据名' ,a.table_name as '表名' ,a.index_name as '索引1' ,b.INDEX_NAME as '索引2' ,a.COLUMN_NAME as '重复列名' from statistics b on a.table_schema=b.table_schema and a.table_name=b.table_name and a.seq_in_index=b.seq_in_index and a.column_name = b.column_name where a.seq_in_index = 1 and a.index_name<>b.index_name show create table employees.dept_emp;
